#include <iostream>
#include <algorithm>
#include <map>
#include <vector>
#define INF 99999999
using namespace std;

int n;
int mp, mf, ms, mv; // 최소 단백질, 지방, 탄수화물, 비타민
int b, c, d, e, sum = 0;
int ret = INF;

struct _ {
	int mp, mf, ms, mv, cost;
}arr[16];
int main() {
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	map<int, vector<vector<int> >> v_v;
	cin >> n;
	cin >> mp >> mf >> ms >> mv;
	for (int i = 0; i < n; i++) {
		cin >> arr[i].mp >> arr[i].mf >> arr[i].ms >> arr[i].mv >> arr[i].cost;
	}

	for (int i = 1; i < (1 << n); i++) {
		b = c = d = e = sum = 0;
		vector<int> v;
		for (int j = 0; j < n; j++) {
			if (i & (1 << j)) {
				v.push_back(j + 1);
				b += arr[j].mp;
				c += arr[j].mf;
				d += arr[j].ms;
				e += arr[j].mv;
				sum += arr[j].cost;
			}
		}
		if (b >= mp && c >= mf && d >= ms && e >= mv) {
			if (ret >= sum) {
				ret = sum;
				v_v[ret].push_back(v);
			}
		}
	}
	sort(v_v[ret].begin(), v_v[ret].end());
	if (ret == INF) { cout << -1 << '\n'; }
	else {
		cout << ret << '\n';
		//for (int a : v_v[ret][0]) { cout << a << " "; }
		for (int i = 0; i < v_v[ret][0].size(); i++) {
			cout << v_v[ret][0][i] << " ";
		}
		cout << '\n';
	}

	return 0;
}
